#MESSAGE("start build")	# 打印信息

SET(TARGET_NAME "test_main")
PROJECT(${TARGET_NAME})

# 头文件路径添加
INCLUDE_DIRECTORIES(${CMAKE_CURRENT_SOURCE_DIR})
INCLUDE_DIRECTORIES(${COMMON_OUTPUT_PATH}/inc)

# 源码文件列表添加
AUX_SOURCE_DIRECTORY(. BIN_SRC_LIST)
#SET(BIN_SRC_LIST test_main.c)

# 库路径添加
LINK_DIRECTORIES(${COMMON_OUTPUT_PATH}/lib)

# 添加子目录
ADD_SUBDIRECTORY(src)

# 执行文件生成
ADD_EXECUTABLE(${TARGET_NAME} ${BIN_SRC_LIST})

# 库链接
TARGET_LINK_LIBRARIES(${TARGET_NAME} hello_world)
